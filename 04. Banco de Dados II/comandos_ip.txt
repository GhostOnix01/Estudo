# Como saber o nome das interfaces
ls /sys/class/net/

# Exibir as redes disponiveis 
ip link show

# Ver IPs das interfaces de redes
ip addr show
# ou 
ip a

# Ver configuraçoes de 
#IPV4
ip -4 a
#IPV6
ip -6 a

# Iniciar a placa
ip link set enp0s3 up

# Derrubar a placa
ip link set enp0s3 down

# Ver o IP de uma interface
ip addr ls enp0s3
#ou
ip addr show enp0s3

# Estatistica
ip -s link show enp0s3

# atribuir manual um IP
ip addr add 192.168.12.100/24 dev enp0s3

# Excluir um IP
ip addr del 192.168.12.100/24 dev enp0s3

# Verificar roteamento
ip route show

#Adicionar Rotas
ip route add 10.20.30.0/24 via 192.168.100.10 dev enp0s3

# Mudar o nome da interface para o padrão antigo
ip link set enp0s3 name eth0
